import json

from dataclasses import dataclass, field
from typing import Any, Optional

import anndata as ad


@dataclass
class ReaderResult:
    """Uniform output from every reader module."""
    adata: ad.AnnData
    source_meta: dict = field(default_factory=dict)
    # source_meta keys:
    #   source_format: str (e.g. "seurat_rds", "csv", "mtx")
    #   reader_used: str (e.g. "seurat_reader")
    #   matrix_orientation_before: str ("cells_x_genes" | "genes_x_cells")
    #   transposed: bool
    #   assay_selected: str | None
    #   layer_selected: str | None
    #   raw_counts_found: bool
    #   feature_types_present: list[str]
    #   matrix_type_hint: str | None  (trusted label from source, e.g. "counts")
    #   decompressed: bool
    #   original_filename: str
    #   warnings: list[str]


class DecisionNeeded(Exception):
    """Raised when a reader or pipeline step cannot proceed without user input."""

    def __init__(
        self,
        decision_type: str,
        context: str,
        options: list[str],
        recommendation: Optional[str] = None,
        reason: Optional[str] = None,
        partial_state: Optional[str] = None,
    ):
        self.decision_type = decision_type
        self.context = context
        self.options = options
        self.recommendation = recommendation
        self.reason = reason
        self.partial_state = partial_state
        super().__init__(f"Decision needed: {decision_type}")

    def to_json(self) -> str:
        return json.dumps({
            "status": "decision_needed",
            "decision_type": self.decision_type,
            "context": self.context,
            "options": self.options,
            "recommendation": self.recommendation,
            "reason": self.reason,
            "partial_state": self.partial_state,
        }, indent=2)

import gzip

from typing import Optional

def sample_matrix(X, n_sample: int = 1000, seed: int = 42) -> np.ndarray:
    """
    Sample rows from a matrix (sparse or dense) and return dense array.

    Args:
        X: Matrix (sparse or dense), shape (n_obs, n_vars).
        n_sample: Max number of rows to sample.
        seed: Random seed.

    Returns:
        Dense numpy array of shape (min(n_sample, n_obs), n_vars).
    """
    n_obs = X.shape[0]
    rng = np.random.default_rng(seed)
    idx = rng.choice(n_obs, size=min(n_sample, n_obs), replace=False)
    idx.sort()
    if sparse.issparse(X):
        return X[idx, :].toarray()
    return np.asarray(X[idx, :])


def is_integer_like(arr: np.ndarray, tol: float = 1e-6) -> bool:
    """Check if all values in array are approximately integer."""
    if arr.size == 0:
        return True
    return bool(np.allclose(arr, np.round(arr), atol=tol))


def make_names_unique(names: list[str]) -> tuple[list[str], dict[str, str]]:
    """
    Make a list of names unique by appending suffixes.

    Returns:
        (unique_names, rename_map) where rename_map maps new→original
        only for names that were changed.
    """
    counts: dict[str, int] = {}
    result = []
    rename_map = {}
    for name in names:
        if name in counts:
            counts[name] += 1
            new_name = f"{name}-{counts[name]}"
            result.append(new_name)
            rename_map[new_name] = name
        else:
            counts[name] = 0
            result.append(name)
    return result, rename_map


def format_number(n: int) -> str:
    """Format integer with commas: 28686 -> '28,686'."""
    return f"{n:,}"


def format_size(size_bytes: int) -> str:
    """Format byte count as human-readable size."""
    for unit in ["B", "KB", "MB", "GB", "TB"]:
        if size_bytes < 1024:
            return f"{size_bytes:.1f} {unit}"
        size_bytes /= 1024
    return f"{size_bytes:.1f} PB"


def iso_now() -> str:
    """Return current UTC time as ISO 8601 string."""
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


# ---------------------------------------------------------------------------
# Compression / archive helpers
# ---------------------------------------------------------------------------

ARCHIVE_EXTENSIONS = {".tar.gz", ".tgz", ".tar.bz2", ".tbz2", ".tar", ".zip"}


def strip_compression_ext(path: str) -> tuple[str, bool]:
    """
    Strip compression extension from path for format detection.

    Returns:
        (stripped_path, was_compressed)
    """
    p = Path(path)
    name = p.name

    # Multi-part extensions first
    for ext in (".tar.gz", ".tar.bz2"):
        if name.endswith(ext):
            return str(p.with_name(name[: -len(ext)])), True

    if p.suffix in (".gz", ".bz2"):
        return str(p.with_suffix("")), True

    return path, False


def is_archive(path: str) -> bool:
    """Check if path is a recognized archive format."""
    p = Path(path)
    name = p.name.lower()
    for ext in ARCHIVE_EXTENSIONS:
        if name.endswith(ext):
            return True
    return False


def extract_archive(path: str, dest: Optional[str] = None) -> str:
    """
    Extract archive to a temp directory.

    Returns:
        Path to the directory containing extracted contents.
    """
    if dest is None:
        dest = tempfile.mkdtemp(prefix="stanobj_archive_")

    p = Path(path)
    name = p.name.lower()

    if name.endswith((".tar.gz", ".tgz", ".tar.bz2", ".tbz2", ".tar")):
        with tarfile.open(path, "r:*") as tf:
            tf.extractall(dest, filter="data")
    elif name.endswith(".zip"):
        with zipfile.ZipFile(path, "r") as zf:
            zf.extractall(dest)
    else:
        raise ValueError(f"Unsupported archive format: {path}")

    return dest


def decompress_to_temp(path: str) -> str:
    """
    Decompress a single .gz or .bz2 file to a temp file.

    Returns:
        Path to decompressed temp file.
    """
    p = Path(path)
    if p.suffix == ".gz":
        opener = gzip.open
    elif p.suffix == ".bz2":
        opener = bz2.open
    else:
        raise ValueError(f"Not a compressed file: {path}")

    stem = p.stem  # e.g. "data.h5ad" from "data.h5ad.gz"
    tmp = tempfile.NamedTemporaryFile(
        prefix="stanobj_", suffix=f"_{stem}", delete=False
    )
    with opener(path, "rb") as f_in, open(tmp.name, "wb") as f_out:
        shutil.copyfileobj(f_in, f_out)
    return tmp.name


def cleanup_temp(path: str) -> None:
    """Remove a temp file or directory, silently ignoring errors."""
    p = Path(path)
    if p.is_dir():
        shutil.rmtree(p, ignore_errors=True)
    elif p.is_file():
        p.unlink(missing_ok=True)

def detect_format(path: str) -> str:
    """
    Detect input format from file extension, directory contents, and HDF5 structure.

    Args:
        path: Input file or directory path.

    Returns:
        One of: "h5ad", "seurat_rds", "csv", "tsv", "mtx", "10x_h5",
                "generic_h5", "loom"

    Raises:
        ValueError: If format cannot be determined.
    """
    p = Path(path)

    # Directory: check for MTX triplet
    if p.is_dir():
        contents = [f.name.lower() for f in p.iterdir()]
        has_mtx = any(f.startswith("matrix.mtx") for f in contents)
        if has_mtx:
            return "mtx"
        raise ValueError(
            f"Cannot detect format: directory '{path}' has no matrix.mtx file"
        )

    # Strip compression for extension-based detection
    stripped, _ = strip_compression_ext(str(p))
    ext = Path(stripped).suffix.lower()

    # Extension-based detection
    if ext == ".h5ad":
        return "h5ad"
    if ext == ".rds":
        return "seurat_rds"
    if ext == ".csv":
        return "csv"
    if ext in (".tsv", ".txt"):
        return "tsv"
    if ext == ".loom":
        return "loom"
    if ext == ".mtx":
        # Single mtx file: look for sibling barcodes/features
        return "mtx"

    # HDF5 files need content inspection
    if ext in (".h5", ".hdf5"):
        return _inspect_h5(path)

    raise ValueError(
        f"Cannot detect format for '{path}' (extension: '{ext}'). "
        "Use --format to specify explicitly."
    )


def _inspect_h5(path: str) -> str:
    """Inspect HDF5 file to determine if it's 10x, h5ad-like, or generic."""
    import h5py

    # May need to decompress first
    actual_path = path
    needs_cleanup = False
    p = Path(path)
    if p.suffix in (".gz", ".bz2"):
        from utils import decompress_to_temp
        actual_path = decompress_to_temp(path)
        needs_cleanup = True

    try:
        with h5py.File(actual_path, "r") as f:
            keys = set(f.keys())
            # 10x format: has "matrix" group
            if "matrix" in keys:
                return "10x_h5"
            # h5ad-like: has "X" or "obs" groups
            if "X" in keys or "obs" in keys:
                return "h5ad"
            return "generic_h5"
    finally:
        if needs_cleanup:
            os.unlink(actual_path)

def compute_evidence(X) -> dict:
    """
    Compute evidence metrics from a matrix (sparse or dense).

    Args:
        X: Matrix of shape (n_obs, n_vars), sparse or dense.

    Returns:
        dict of evidence metrics.
    """
    if sparse.issparse(X):
        X_dense = X.toarray()
    else:
        X_dense = np.asarray(X)

    flat = X_dense.flatten()
    non_zero = flat[flat != 0]

    has_negatives = bool(np.any(X_dense < 0))
    min_value = float(np.min(X_dense)) if X_dense.size > 0 else 0.0
    max_value = float(np.max(X_dense)) if X_dense.size > 0 else 0.0
    mean_value = float(np.mean(X_dense)) if X_dense.size > 0 else 0.0
    sparsity = float(np.sum(X_dense == 0) / X_dense.size) if X_dense.size > 0 else 1.0
    is_int = is_integer_like(non_zero) if len(non_zero) > 0 else True

    # Row sums CV
    row_sums = np.sum(X_dense, axis=1)
    row_mean = float(np.mean(row_sums)) if len(row_sums) > 0 else 0.0
    row_std = float(np.std(row_sums)) if len(row_sums) > 0 else 0.0
    row_sums_cv = row_std / row_mean if row_mean > 0 else 0.0

    # Column (gene) stats for scaled detection
    if X_dense.shape[1] > 0 and X_dense.shape[0] > 1:
        col_means = np.mean(X_dense, axis=0)
        col_stds = np.std(X_dense, axis=0)
        near_zero_mean_frac = float(np.sum(np.abs(col_means) < 0.5) / len(col_means))
        near_one_std_frac = float(np.sum(np.abs(col_stds - 1) < 0.5) / len(col_stds))
    else:
        near_zero_mean_frac = 0.0
        near_one_std_frac = 0.0

    return {
        "has_negatives": has_negatives,
        "is_integer": is_int,
        "min_value": round(min_value, 4),
        "max_value": round(max_value, 4),
        "mean_value": round(mean_value, 4),
        "sparsity": round(sparsity, 4),
        "row_sums_cv": round(row_sums_cv, 4),
        "near_zero_mean_frac": round(near_zero_mean_frac, 4),
        "near_one_std_frac": round(near_one_std_frac, 4),
        "n_cells_sampled": X_dense.shape[0],
        "n_genes": X_dense.shape[1],
    }


def classify_matrix_type(evidence: dict) -> tuple[str, str]:
    """
    Classify matrix type from evidence metrics.

    Returns:
        (matrix_type, confidence) where matrix_type is one of:
        "counts", "normalized", "log1p", "scaled", "unknown"
        and confidence is "high", "medium", or "low".
    """
    has_neg = evidence["has_negatives"]
    is_int = evidence["is_integer"]
    max_val = evidence["max_value"]
    sparsity = evidence["sparsity"]
    row_cv = evidence["row_sums_cv"]
    near_zero = evidence["near_zero_mean_frac"]
    near_one = evidence["near_one_std_frac"]

    # Rule 1: Scaled (z-score) — has negatives, per-gene mean ~0 and std ~1
    if has_neg and near_zero > 0.7 and near_one > 0.7:
        return "scaled", "high"

    # Rule 2: Raw counts — non-negative integers, high sparsity, high max
    if not has_neg and is_int and max_val > 10:
        confidence = "high" if sparsity > 0.8 else "medium"
        return "counts", confidence

    # Rule 3: Normalized (not log) — non-negative floats, very consistent row sums
    if not has_neg and not is_int and row_cv < 0.01 and max_val > 100:
        return "normalized", "high"

    # Rule 4: Log-normalized — non-negative floats, max < 20
    if not has_neg and not is_int and max_val < 20:
        confidence = "high" if max_val < 15 else "medium"
        return "log1p", confidence

    # Rule 5: Possibly normalized with some row-sum variation
    if not has_neg and not is_int and row_cv < 0.1:
        return "normalized", "low"

    # Fallback: negative but doesn't match scaled pattern
    if has_neg:
        return "scaled", "low"

    return "unknown", "low"


def classify_with_override(
    evidence: dict,
    source_hint: Optional[str] = None,
) -> dict:
    """
    Classify matrix type, applying source metadata override when available.

    Args:
        evidence: Evidence dict from compute_evidence.
        source_hint: Trusted label from source format (e.g. "counts" from
                     Seurat counts slot). None means no override.

    Returns:
        {
            "matrix_type": str,
            "confidence": str,
            "matrix_type_source": "heuristic" | "source_metadata",
            "warnings": list[str],
        }
    """
    heuristic_type, heuristic_conf = classify_matrix_type(evidence)
    warnings = []

    if source_hint is None:
        return {
            "matrix_type": heuristic_type,
            "confidence": heuristic_conf,
            "matrix_type_source": "heuristic",
            "warnings": warnings,
        }

    # Normalize source hint
    hint_map = {
        "counts": "counts",
        "raw_counts": "counts",
        "data": "log1p",
        "logcounts": "log1p",
        "normcounts": "normalized",
        "scale.data": "scaled",
    }
    normalized_hint = hint_map.get(source_hint, source_hint)

    if normalized_hint == heuristic_type:
        return {
            "matrix_type": normalized_hint,
            "confidence": "high",
            "matrix_type_source": "source_metadata",
            "warnings": warnings,
        }

    # Mismatch: trust source but warn
    warnings.append(
        f"Source metadata says '{source_hint}' (→ {normalized_hint}) "
        f"but heuristics suggest '{heuristic_type}' "
        f"({heuristic_conf} confidence). Trusting source metadata."
    )
    return {
        "matrix_type": normalized_hint,
        "confidence": "medium",
        "matrix_type_source": "source_metadata",
        "warnings": warnings,
    }

"""Validation checks for canonical AnnData objects."""

from dataclasses import dataclass, field

import anndata as ad
import numpy as np
from scipy import sparse

try:
    from .utils import sample_matrix, is_integer_like
except ImportError:
    from utils import sample_matrix, is_integer_like


@dataclass
class ValidationResult:
    """Outcome of validate_adata."""
    passed: bool = True
    fatal_errors: list[str] = field(default_factory=list)
    warnings: list[str] = field(default_factory=list)

    def add_fatal(self, msg: str):
        self.fatal_errors.append(msg)
        self.passed = False

    def add_warning(self, msg: str):
        self.warnings.append(msg)


def validate_adata(adata: ad.AnnData, matrix_type: str) -> ValidationResult:
    """
    Run all structural, identifier, semantic, and consistency checks.

    Args:
        adata: AnnData to validate.
        matrix_type: Declared matrix type ("counts", "normalized", etc.).

    Returns:
        ValidationResult with passed flag, fatal errors, and warnings.
    """
    result = ValidationResult()

    # --- Fatal: structural ---
    if adata.X is None:
        result.add_fatal("X matrix is None")
        return result

    ndim = 2
    if sparse.issparse(adata.X):
        ndim = len(adata.X.shape)
    else:
        ndim = adata.X.ndim
    if ndim != 2:
        result.add_fatal(f"X matrix is not 2D (ndim={ndim})")

    if adata.n_obs == 0:
        result.add_fatal("n_obs is 0 — no cells")

    if adata.n_vars == 0:
        result.add_fatal("n_vars is 0 — no genes")

    if adata.X.shape[0] != len(adata.obs_names):
        result.add_fatal(
            f"Matrix rows ({adata.X.shape[0]}) != obs_names length ({len(adata.obs_names)})"
        )

    if adata.X.shape[1] != len(adata.var_names):
        result.add_fatal(
            f"Matrix cols ({adata.X.shape[1]}) != var_names length ({len(adata.var_names)})"
        )

    # --- Fatal: identifier uniqueness ---
    if len(set(adata.obs_names)) != len(adata.obs_names):
        n_dup = len(adata.obs_names) - len(set(adata.obs_names))
        result.add_fatal(f"obs_names are not unique ({n_dup} duplicates)")

    if len(set(adata.var_names)) != len(adata.var_names):
        n_dup = len(adata.var_names) - len(set(adata.var_names))
        result.add_fatal(f"var_names are not unique ({n_dup} duplicates)")

    # Stop here if there are fatal errors — remaining checks may be meaningless
    if not result.passed:
        return result

    # --- Warning: counts declared but not integer-like ---
    if matrix_type == "counts" and adata.n_obs > 0 and adata.n_vars > 0:
        sample = sample_matrix(adata.X, n_sample=200, seed=42)
        non_zero = sample[sample != 0]
        if len(non_zero) > 0 and not is_integer_like(non_zero):
            result.add_warning(
                "Matrix declared as counts but contains non-integer values"
            )
        if np.any(sample < 0):
            result.add_warning(
                "Matrix declared as counts but contains negative values"
            )

    # --- Warning: empty metadata ---
    if len(adata.obs.columns) == 0:
        result.add_warning("No cell metadata columns found in obs")

    if len(adata.var.columns) == 0:
        result.add_warning("No gene metadata columns found in var")

    # --- Warning: extreme sparsity ---
    if adata.n_obs > 0 and adata.n_vars > 0:
        if sparse.issparse(adata.X):
            nnz = adata.X.nnz
        else:
            nnz = np.count_nonzero(adata.X)
        total = adata.n_obs * adata.n_vars
        sparsity = 1.0 - (nnz / total) if total > 0 else 1.0
        if sparsity > 0.999:
            result.add_warning(
                f"Extremely high sparsity ({sparsity:.4%}) — matrix may be nearly empty"
            )

    # --- Warning: small dataset ---
    if 0 < adata.n_obs < 50:
        result.add_warning(f"Very few cells ({adata.n_obs})")
    if 0 < adata.n_vars < 100:
        result.add_warning(f"Very few genes ({adata.n_vars})")

    # --- Warning: embedding dimension mismatch ---
    bad_keys = []
    for key, arr in adata.obsm.items():
        if arr.shape[0] != adata.n_obs:
            bad_keys.append(key)
            result.add_warning(
                f"Embedding '{key}' in obsm has {arr.shape[0]} rows but n_obs={adata.n_obs} — will be dropped"
            )
    # Actually drop the bad embeddings
    for key in bad_keys:
        del adata.obsm[key]

    # --- Warning: feature type column missing with mixed-looking names ---
    if "feature_type" not in adata.var.columns:
        var_names = list(adata.var_names[:100])
        has_adt_like = any(
            name.startswith(("ADT_", "HTO_", "ab_")) or name.endswith(("_TotalSeqA", "_TotalSeqB", "_TotalSeqC"))
            for name in var_names
        )
        if has_adt_like:
            result.add_warning(
                "No 'feature_type' column but var_names suggest non-RNA features may be present"
            )

    return result

# Mapping: source column name → canonical column name
OBS_COLUMN_MAP = {
    # cell_type variants
    "celltype": "cell_type",
    "CellType": "cell_type",
    "Celltype": "cell_type",
    "cell_type_label": "cell_type",
    "annotation": "cell_type",
    "cluster_label": "cell_type",
    # sample variants
    "orig.ident": "sample",
    "sample_id": "sample",
    "sampleName": "sample",
    "Sample": "sample",
    "SampleID": "sample",
    # donor variants
    "patient": "donor",
    "donor_id": "donor",
    "individual": "donor",
    "Patient": "donor",
    "subject": "donor",
    "subject_id": "donor",
    # condition variants
    "disease": "condition",
    "treatment": "condition",
    "experimental_condition": "condition",
    "Disease": "condition",
    "diagnosis": "condition",
    # batch variants
    "batch_id": "batch",
    "Batch": "batch",
}


def standardize_obs(
    obs: pd.DataFrame,
    dataset_name: str,
    make_unique: bool = False,
) -> pd.DataFrame:
    """
    Standardize cell metadata.

    - Adds cell_id (original barcodes) and dataset columns.
    - Maps common column name variants to canonical names.
    - Optionally makes obs_names unique by prepending dataset name.

    Original columns are never deleted.
    """
    result = obs.copy()

    # Store original cell IDs
    result["cell_id"] = result.index.tolist()
    result["dataset"] = dataset_name

    # Map column name variants → canonical names (only add, never overwrite)
    for source_col, canon_col in OBS_COLUMN_MAP.items():
        if source_col in result.columns and canon_col not in result.columns:
            result[canon_col] = result[source_col]

    # Make obs_names unique if needed
    if make_unique or len(set(result.index)) != len(result.index):
        new_names, _ = make_names_unique(result.index.tolist())
        # If still not unique (e.g. barcodes repeated across samples), prepend dataset
        if len(set(new_names)) != len(new_names):
            new_names = [f"{dataset_name}_{n}" for n in result["cell_id"]]
            new_names, _ = make_names_unique(new_names)
        result.index = pd.Index(new_names)

    return result


def standardize_var(
    var: pd.DataFrame,
    return_rename_map: bool = False,
) -> "pd.DataFrame | tuple[pd.DataFrame, dict]":
    """
    Standardize gene metadata.

    - Ensures gene_symbol column exists.
    - Resolves duplicate var_names with suffix, storing originals.

    Original columns are never deleted.
    """
    result = var.copy()

    # Ensure gene_symbol column
    if "gene_symbol" not in result.columns:
        result["gene_symbol"] = result.index.tolist()

    # Handle duplicates
    names = result.index.tolist()
    rename_map: dict = {}
    if len(set(names)) != len(names):
        new_names, rename_map = make_names_unique(names)
        result["original_gene_name"] = names
        result.index = pd.Index(new_names)

    if return_rename_map:
        return result, rename_map
    return result


def standardize_obsm(obsm: dict) -> dict:
    """
    Rename common embedding keys to canonical names.

    Mapping: pca→X_pca, umap→X_umap, tsne→X_tsne, etc.
    """
    key_map = {
        "pca": "X_pca",
        "PCA": "X_pca",
        "umap": "X_umap",
        "UMAP": "X_umap",
        "tsne": "X_tsne",
        "tSNE": "X_tsne",
    }
    result = {}
    for key, arr in obsm.items():
        new_key = key_map.get(key, key)
        result[new_key] = arr
    return result


def assign_layers(
    adata: ad.AnnData,
    matrix_type: str,
    source_layers: dict,
) -> ad.AnnData:
    """
    Assign X and layers according to canonical conventions.

    Convention:
    - If counts available + processed also available → X=processed, layers["counts"]=counts
    - If counts only → X=counts, layers["counts"]=counts
    - If processed only → X=processed, warn counts missing
    """
    if matrix_type == "counts":
        # X is counts
        adata.layers["counts"] = adata.X.copy()
        # Check if we also have a processed layer to promote to X
        processed_key = None
        for key in ("log1p", "logcounts", "data", "normalized", "normcounts"):
            if key in source_layers:
                processed_key = key
                break
        if processed_key is not None:
            adata.X = source_layers[processed_key]
            adata.uns["stanobj_x_contents"] = processed_key
        else:
            adata.uns["stanobj_x_contents"] = "counts"
    else:
        # X is processed data
        adata.uns["stanobj_x_contents"] = matrix_type
        # Check if counts are in source_layers
        for key in ("counts", "raw_counts"):
            if key in source_layers:
                adata.layers["counts"] = source_layers[key]
                break

    # Copy remaining source layers
    for key, mtx in source_layers.items():
        if key not in adata.layers and key != adata.uns.get("stanobj_x_contents"):
            adata.layers[key] = mtx

    return adata


def add_provenance(
    adata: ad.AnnData,
    source_meta: dict,
    matrix_classification: dict,
    decisions: dict,
    all_warnings: list[str],
) -> ad.AnnData:
    """Add conversion provenance to adata.uns['stanobj']."""
    adata.uns["stanobj"] = {
        "source_path": source_meta.get("source_path", ""),
        "source_format": source_meta.get("source_format", ""),
        "conversion_timestamp": iso_now(),
        "matrix_type": matrix_classification.get("matrix_type", "unknown"),
        "x_contents": adata.uns.get("stanobj_x_contents", "unknown"),
        "transposed": source_meta.get("transposed", False),
        "assay_selected": source_meta.get("assay_selected"),
        "layer_selected": source_meta.get("layer_selected"),
        "modality_filter": source_meta.get("modality_filter"),
        "decompressed": source_meta.get("decompressed", False),
        "obs_name_strategy": source_meta.get("obs_name_strategy", "original"),
        "var_name_strategy": source_meta.get("var_name_strategy", "original"),
        "decisions_made": decisions,
        "warnings": all_warnings,
        "version": "1.0.0",
    }
    # Clean up temp key
    adata.uns.pop("stanobj_x_contents", None)
    return adata


def ensure_sparse(adata: ad.AnnData, threshold: int = 10_000) -> bool:
    """
    Convert dense X to sparse if n_obs exceeds threshold.

    Returns True if conversion happened.
    """
    if not sparse.issparse(adata.X) and adata.n_obs >= threshold:
        adata.X = sparse.csr_matrix(adata.X)
        return True
    return False

def generate_audit_log(
    source_path: str,
    source_format: str,
    output_path: str,
    source_meta: dict,
    matrix_classification: dict,
    x_contents: str,
    n_cells: int,
    n_genes: int,
    decisions: dict,
    warnings: list[str],
) -> str:

    lines = []
    lines.append("=== stanobj conversion audit ===")
    lines.append(f"Source: {source_path} ({source_format})")
    lines.append(f"Output: {output_path}")
    lines.append(f"Timestamp: {iso_now()}")
    lines.append("")

    lines.append("--- Detection ---")
    lines.append(f"Format: {source_format}")
    reader = source_meta.get("reader_used", "unknown")
    lines.append(f"Reader: {reader}")
    if source_meta.get("assay_selected"):

    if source_meta.get("decompressed"):
        lines.append(f"Decompressed from: {source_meta.get('original_filename', 'unknown')}")
    lines.append("")

    lines.append("--- Standardization ---")
    lines.append(f"X contents: {x_contents}")
    orient = source_meta.get("matrix_orientation_before", "unknown")
    transposed = source_meta.get("transposed", False)
    lines.append(f"Orientation: {orient} {'(transposed)' if transposed else '(no transpose)'}")

        lines.append(f"Embeddings preserved: {', '.join(source_meta['embeddings_preserved'])}")
    if source_meta.get("obs_columns_standardized"):
        mappings = ", ".join(
            f"{k} -> {v}" for k, v in source_meta["obs_columns_standardized"].items()
        )
        lines.append(f"Metadata columns mapped: {mappings}")
    lines.append("")

    if decisions:
        lines.append("--- Decisions ---")
        for key, val in decisions.items():
            lines.append(f"[DECISION] {key}: {val}")
        lines.append("")

    lines.append("--- Warnings ---")
    if warnings:
        for w in warnings:
            lines.append(f"[WARN] {w}")
    else:
        lines.append("None")
    lines.append("")

    lines.append("--- Validation ---")
    lines.append("All checks passed.")
    lines.append("")
    lines.append(f"Shape: {format_number(n_cells)} cells x {format_number(n_genes)} genes")

    return "\n".join(lines)

"""Matrix Market / 10x triplet reader for stanobj."""

import os
from pathlib import Path
from typing import Optional

import anndata as ad
import numpy as np
import pandas as pd
from scipy import sparse
from scipy.io import mmread

try:
    from .base import ReaderResult
except ImportError:

def read_mtx(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """
    Read a Matrix Market triplet directory or .mtx file.

    Expects:
      - matrix.mtx[.gz]
      - barcodes.tsv[.gz]
      - features.tsv[.gz] or genes.tsv[.gz]

    MTX convention is genes x cells. This reader transposes to cells x genes.

    Args:
        path: Directory containing the triplet, or path to a .mtx file.
        decisions: Pre-supplied decisions (not used for MTX, reserved for interface).

    Returns:
        ReaderResult with transposed AnnData and source metadata.
    """
    # Resolve directory
    p = Path(path)
    if p.is_file():
        mtx_dir = str(p.parent)
    else:
        mtx_dir = str(p)

    # Find files (handle .gz variants)
    mtx_file = _find_file(mtx_dir, ["matrix.mtx.gz", "matrix.mtx"])
    barcodes_file = _find_file(mtx_dir, ["barcodes.tsv.gz", "barcodes.tsv"])
    features_file = _find_file(
        mtx_dir,
        ["features.tsv.gz", "features.tsv", "genes.tsv.gz", "genes.tsv"],
    )

    if mtx_file is None:
        raise FileNotFoundError(f"No matrix.mtx[.gz] found in {mtx_dir}")
    if barcodes_file is None:
        raise FileNotFoundError(f"No barcodes.tsv[.gz] found in {mtx_dir}")
    if features_file is None:
        raise FileNotFoundError(f"No features.tsv[.gz] or genes.tsv[.gz] found in {mtx_dir}")

    warnings = []

    # Read matrix (genes x cells)
    mtx = mmread(mtx_file)
    mtx = sparse.csr_matrix(mtx)

    # Read barcodes
    barcodes = pd.read_csv(
        barcodes_file, sep="\t", header=None, names=["barcode"]
    )
    cell_ids = barcodes["barcode"].tolist()

    # Read features (2 or 3 columns)
    features = pd.read_csv(features_file, sep="\t", header=None)
    n_cols = features.shape[1]

    gene_ids = features[0].tolist()
    gene_symbols = features[1].tolist() if n_cols >= 2 else gene_ids
    feature_types = features[2].tolist() if n_cols >= 3 else ["Gene Expression"] * len(gene_ids)

    # Validate dimensions (mtx is genes x cells)
    n_genes_mtx, n_cells_mtx = mtx.shape

    if n_cells_mtx != len(cell_ids):
        raise ValueError(
            f"Matrix columns ({n_cells_mtx}) != barcodes count ({len(cell_ids)})"
        )
    if n_genes_mtx != len(gene_ids):
        raise ValueError(
            f"Matrix rows ({n_genes_mtx}) != features count ({len(gene_ids)})"
        )

    # Transpose to cells x genes
    mtx_t = mtx.T.tocsr()

    # Build var DataFrame
    var = pd.DataFrame({
        "gene_id": gene_ids,
        "gene_symbol": gene_symbols,
        "feature_type": feature_types,
    }, index=gene_symbols)

    # Build obs DataFrame
    obs = pd.DataFrame(index=cell_ids)

    # Create AnnData
    adata = ad.AnnData(X=mtx_t, obs=obs, var=var)

    # Detect feature types present
    ft_present = list(set(feature_types))

    source_meta = {
        "source_format": "mtx",
        "reader_used": "mtx_reader",
        "matrix_orientation_before": "genes_x_cells",
        "transposed": True,
        "raw_counts_found": True,  # MTX from 10x is typically raw counts
        "feature_types_present": ft_present,
        "matrix_type_hint": "counts",
        "decompressed": any(
            f.endswith(".gz")
            for f in [mtx_file, barcodes_file, features_file]
        ),
        "warnings": warnings,
    }

    return ReaderResult(adata=adata, source_meta=source_meta)


def _find_file(directory: str, candidates: list[str]) -> Optional[str]:
    """Find the first existing file from a list of candidates in a directory."""
    for name in candidates:
        full = os.path.join(directory, name)
        if os.path.isfile(full):
            return full
    return None

```python
"""CSV/TSV reader for stanobj. The most ambiguous input format."""

import re
from pathlib import Path
from typing import Optional

import anndata as ad
import numpy as np
import pandas as pd
from scipy import sparse

try:
    from .base import ReaderResult, DecisionNeeded
except ImportError:
    from base import ReaderResult, DecisionNeeded


# Common gene name patterns
_GENE_PATTERNS = [
    re.compile(r"^ENS[A-Z]*G\d+"),          # Ensembl
    re.compile(r"^[A-Z][A-Z0-9\-]{1,15}$"),  # Human symbol (TP53, BRCA1)
    re.compile(r"^[A-Z][a-z0-9\-]{1,15}$"),  # Mouse symbol (Trp53)
]

# Common barcode patterns
_BARCODE_PATTERNS = [
    re.compile(r"^[ACGT]{8,}"),              # 10x-style barcodes
    re.compile(r"^[A-Za-z0-9]+[-_]\d+$"),    # custom: prefix-number
]


def read_csv(
    path: str,
    delimiter: Optional[str] = None,
    decisions: Optional[dict] = None,
) -> ReaderResult:
    """
    Read a CSV or TSV expression matrix.

    Detects orientation (cells x genes vs genes x cells) using heuristics.
    Raises DecisionNeeded if orientation is ambiguous.

    Args:
        path: Path to .csv, .tsv, .csv.gz, .tsv.gz file.
        delimiter: Explicit delimiter. If None, inferred from extension.
        decisions: Pre-supplied decisions dict.

    Returns:
        ReaderResult with AnnData in cells x genes orientation.
    """
    decisions = decisions or {}

    # Infer delimiter from extension
    if delimiter is None:
        ext = Path(path).suffix.lower()
        if ext == ".gz":
            ext = Path(Path(path).stem).suffix.lower()
        delimiter = "\t" if ext in (".tsv", ".txt") else ","

    # Read the full table (index_col=0 assumes first column is identifiers)
    df = pd.read_csv(path, sep=delimiter, index_col=0)

    row_labels = list(df.index.astype(str))
    col_labels = list(df.columns.astype(str))

    warnings = []

    # Determine orientation
    orientation = decisions.get("matrix_orientation")
    transposed = False

    if orientation is None:
        orientation = _infer_orientation(row_labels, col_labels, df.shape)

    if orientation == "cells_x_genes":
        transposed = False
    elif orientation == "genes_x_cells":
        df = df.T
        transposed = True
        warnings.append("Matrix was transposed from genes x cells to cells x genes")
    else:
        # Ambiguous — raise decision
        raise DecisionNeeded(
            decision_type="matrix_orientation",
            context=(
                f"CSV has shape {df.shape}. "
                f"Row labels sample: {row_labels[:3]}. "
                f"Column labels sample: {col_labels[:3]}. "
                f"Cannot determine if rows are cells or genes."
            ),
            options=["cells_x_genes", "genes_x_cells"],
            recommendation=None,
            reason="Neither row nor column labels match known gene or barcode patterns",
        )

    # Convert to sparse
    X = sparse.csr_matrix(df.values.astype(np.float32))

    obs = pd.DataFrame(index=df.index.astype(str))
    var = pd.DataFrame(index=df.columns.astype(str))

    adata = ad.AnnData(X=X, obs=obs, var=var)

    source_meta = {
        "source_format": "csv" if delimiter == "," else "tsv",
        "reader_used": "csv_reader",
        "matrix_orientation_before": "genes_x_cells" if transposed else "cells_x_genes",
        "transposed": transposed,
        "decompressed": path.endswith(".gz"),
        "warnings": warnings,
    }

    return ReaderResult(adata=adata, source_meta=source_meta)


def _infer_orientation(
    row_labels: list[str],
    col_labels: list[str],
    shape: tuple[int, int],
) -> str:
    """
    Infer whether rows are cells or genes.

    Returns: "cells_x_genes", "genes_x_cells", or "ambiguous"
    """
    n_rows, n_cols = shape

    rows_gene_like = _fraction_matching(row_labels[:100], _GENE_PATTERNS)
    cols_gene_like = _fraction_matching(col_labels[:100], _GENE_PATTERNS)
    rows_barcode_like = _fraction_matching(row_labels[:100], _BARCODE_PATTERNS)
    cols_barcode_like = _fraction_matching(col_labels[:100], _BARCODE_PATTERNS)

    # Strong signal: row labels are genes, col labels are barcodes
    if rows_gene_like > 0.5 and cols_barcode_like > 0.3:
        return "genes_x_cells"

    # Strong signal: col labels are genes, row labels are barcodes
    if cols_gene_like > 0.5 and rows_barcode_like > 0.3:
        return "cells_x_genes"

    # Weaker signal: just gene-like labels
    if rows_gene_like > 0.5 and cols_gene_like < 0.2:
        return "genes_x_cells"
    if cols_gene_like > 0.5 and rows_gene_like < 0.2:
        return "cells_x_genes"

    # Shape heuristic: typical gene counts are 15k-60k
    if n_cols > 10_000 and n_rows < n_cols:
        return "cells_x_genes"
    if n_rows > 10_000 and n_cols < n_rows:
        return "genes_x_cells"

    return "ambiguous"


def _fraction_matching(labels: list[str], patterns: list[re.Pattern]) -> float:
    """Fraction of labels matching any of the given regex patterns."""
    if not labels:
        return 0.0
    count = sum(
        1 for label in labels
        if any(p.match(label) for p in patterns)
    )
    return count / len(labels)

import os
from typing import Optional

import anndata as ad

def read_10x_h5(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """
    Read a 10x Genomics HDF5 file (contains matrix/ group).

    The internal format stores genes x cells in CSC format.
    This reader transposes to cells x genes.
    """
    warnings = []

    with h5py.File(path, "r") as f:
        grp = f["matrix"]

        # Read sparse matrix components
        data = grp["data"][:]
        indices = grp["indices"][:]
        indptr = grp["indptr"][:]
        shape = tuple(grp["shape"][:])  # (n_genes, n_cells)

        # Reconstruct CSC matrix (genes x cells)
        mtx = sparse.csc_matrix((data, indices, indptr), shape=shape)

        # Barcodes
        barcodes_raw = grp["barcodes"][:]
        cell_ids = [b.decode("utf-8") if isinstance(b, bytes) else str(b) for b in barcodes_raw]

        # Features
        feat = grp["features"]
        gene_ids = [x.decode("utf-8") if isinstance(x, bytes) else str(x) for x in feat["id"][:]]
        gene_symbols = [x.decode("utf-8") if isinstance(x, bytes) else str(x) for x in feat["name"][:]]

        feature_types = []
        if "feature_type" in feat:
            feature_types = [x.decode("utf-8") if isinstance(x, bytes) else str(x) for x in feat["feature_type"][:]]
        else:
            feature_types = ["Gene Expression"] * len(gene_ids)

    # Transpose to cells x genes
    mtx_t = mtx.T.tocsr()

    var = pd.DataFrame({
        "gene_id": gene_ids,
        "gene_symbol": gene_symbols,
        "feature_type": feature_types,
    }, index=gene_symbols)

    obs = pd.DataFrame(index=cell_ids)

    adata = ad.AnnData(X=mtx_t, obs=obs, var=var)

    source_meta = {
        "source_format": "10x_h5",
        "reader_used": "h5_reader:10x",
        "matrix_orientation_before": "genes_x_cells",
        "transposed": True,
        "raw_counts_found": True,
        "feature_types_present": list(set(feature_types)),
        "matrix_type_hint": "counts",
        "decompressed": False,
        "warnings": warnings,
    }

    return ReaderResult(adata=adata, source_meta=source_meta)


def read_generic_h5(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """
    Attempt to read a generic (non-10x, non-h5ad) HDF5 file.

    Since structure is unknown, inspect and raise DecisionNeeded with
    the HDF5 tree so the user/Claude can decide how to proceed.
    """
    tree_lines = []

    def _visitor(name, obj):
        if isinstance(obj, h5py.Dataset):
            tree_lines.append(f"  Dataset: {name}  shape={obj.shape}  dtype={obj.dtype}")
        elif isinstance(obj, h5py.Group):
            tree_lines.append(f"  Group: {name}/")

    with h5py.File(path, "r") as f:
        f.visititems(_visitor)

    tree_str = "\n".join(tree_lines[:50])  # cap for readability
    if len(tree_lines) > 50:
        tree_str += f"\n  ... ({len(tree_lines) - 50} more items)"

    raise DecisionNeeded(
        decision_type="format_detection",
        context=(
            f"HDF5 file '{path}' has an unrecognized structure.\n"
            f"Contents:\n{tree_str}"
        ),
        options=["10x_h5", "h5ad", "skip"],
        recommendation=None,
        reason="No recognized groups (matrix/, X/, obs/) found",
    )

```python
"""Loom format reader for stanobj."""

from typing import Optional

import anndata as ad
import h5py
import numpy as np
import pandas as pd
from scipy import sparse

try:
    from .base import ReaderResult
except ImportError:
    from base import ReaderResult


def read_loom(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """
    Read a loom file. Loom stores matrices as genes x cells.
    This reader transposes to cells x genes.

    Uses h5py directly for robustness (loompy can be finicky).
    """
    warnings = []

    with h5py.File(path, "r") as f:
        # Main matrix: genes x cells
        matrix = f["matrix"][:]

        # Row attributes (genes)
        row_attrs = {}
        if "row_attrs" in f:
            for key in f["row_attrs"]:
                vals = f["row_attrs"][key][:]
                if vals.dtype.kind == "S" or vals.dtype.kind == "O":
                    vals = np.array([v.decode("utf-8") if isinstance(v, bytes) else str(v) for v in vals])
                row_attrs[key] = vals

        # Column attributes (cells)
        col_attrs = {}
        if "col_attrs" in f:
            for key in f["col_attrs"]:
                vals = f["col_attrs"][key][:]
                if vals.dtype.kind == "S" or vals.dtype.kind == "O":
                    vals = np.array([v.decode("utf-8") if isinstance(v, bytes) else str(v) for v in vals])
                col_attrs[key] = vals

    n_genes, n_cells = matrix.shape

    # Transpose to cells x genes
    X = sparse.csr_matrix(matrix.T.astype(np.float32))

    # Build var from row_attrs
    gene_names = row_attrs.pop("Gene", row_attrs.pop("gene", np.arange(n_genes).astype(str)))
    var = pd.DataFrame(row_attrs, index=gene_names if len(gene_names) == n_genes else range(n_genes))

    # Build obs from col_attrs
    cell_ids = col_attrs.pop("CellID", col_attrs.pop("cellid",
                col_attrs.pop("cell_id", np.arange(n_cells).astype(str))))
    obs = pd.DataFrame(col_attrs, index=cell_ids if len(cell_ids) == n_cells else range(n_cells))

    adata = ad.AnnData(X=X, obs=obs, var=var)

    source_meta = {
        "source_format": "loom",
        "reader_used": "loom_reader",
        "matrix_orientation_before": "genes_x_cells",
        "transposed": True,
        "decompressed": False,
        "warnings": warnings,
    }

    return ReaderResult(adata=adata, source_meta=source_meta)

```python
"""Python bridge for R-based readers (Seurat, SingleCellExperiment)."""

import json
import os
import shutil
import subprocess
import tempfile
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

try:
    from .base import ReaderResult, DecisionNeeded
    from .mtx_reader import read_mtx
except ImportError:
    from base import ReaderResult, DecisionNeeded
    from mtx_reader import read_mtx


_SCRIPTS_DIR = Path(__file__).parent


def read_seurat(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """
    Read a Seurat RDS file via R subprocess.

    Two-phase protocol:
    1. R script inventories the object (assays, slots, reductions).
    2. If multiple assays, either use pre-supplied decision or raise DecisionNeeded.
    3. R script exports selected assay to temp directory as MTX + metadata.
    4. Python reads the exported files.

    Falls back to manual instructions if Rscript is not available.
    """
    decisions = decisions or {}

    # Check R availability
    rscript = shutil.which("Rscript")
    if rscript is None:
        return _r_fallback_instructions(path, "Seurat")

    r_script = str(_SCRIPTS_DIR / "seurat_reader.R")
    tmp_dir = tempfile.mkdtemp(prefix="stanobj_seurat_")

    # Build command
    cmd = [rscript, r_script, path, tmp_dir]
    assay_decision = decisions.get("assay_selection")
    if assay_decision:
        cmd.extend(["--assay", assay_decision])

    try:
        proc = subprocess.run(
            cmd, capture_output=True, text=True, timeout=600
        )

        # Parse JSON output
        stdout = proc.stdout.strip()
        if not stdout:
            raise RuntimeError(f"R script produced no output. stderr: {proc.stderr}")

        result_json = json.loads(stdout)

        if result_json.get("status") == "error":
            # Check if it's actually a SCE
            detected = result_json.get("detected_class", "")
            if "SingleCellExperiment" in detected:
                shutil.rmtree(tmp_dir, ignore_errors=True)
                return read_sce(path, decisions)
            raise RuntimeError(result_json["message"])

        if result_json.get("status") == "decision_needed":
            raise DecisionNeeded(
                decision_type=result_json["decision_type"],
                context=json.dumps(result_json, indent=2),
                options=result_json["options"],
                recommendation=result_json.get("recommendation"),
                reason=result_json.get("reason"),
                partial_state=tmp_dir,
            )

        # Success: read exported files from tmp_dir
        mtx_result = read_mtx(tmp_dir)
        adata = mtx_result.adata

        # Merge metadata from exported CSV
        meta_path = os.path.join(tmp_dir, "metadata.csv")
        if os.path.isfile(meta_path):
            meta = pd.read_csv(meta_path, index_col=0)
            # Align indices
            if len(meta) == adata.n_obs:
                for col in meta.columns:
                    adata.obs[col] = meta[col].values

        # Merge reductions
        for fname in os.listdir(tmp_dir):
            if fname.startswith("reduction_") and fname.endswith(".csv"):
                red_name = fname.replace("reduction_", "").replace(".csv", "")
                emb = pd.read_csv(os.path.join(tmp_dir, fname), index_col=0)
                key = f"X_{red_name}"
                if emb.shape[0] == adata.n_obs:
                    adata.obsm[key] = emb.values

        # Determine matrix type hint from slot used
        slot_used = result_json.get("slot_used", "counts")
        hint_map = {"counts": "counts", "data": "log1p", "scale.data": "scaled"}
        matrix_type_hint = hint_map.get(slot_used, None)

        source_meta = {
            "source_format": "seurat_rds",
            "reader_used": "seurat_reader",
            "matrix_orientation_before": "genes_x_cells",
            "transposed": True,
            "assay_selected": result_json.get("assay"),
            "layer_selected": slot_used,
            "raw_counts_found": slot_used == "counts",
            "matrix_type_hint": matrix_type_hint,
            "feature_types_present": mtx_result.source_meta.get("feature_types_present", []),
            "decompressed": False,
            "reductions_exported": result_json.get("reductions_exported", []),
            "warnings": [],
        }

        return ReaderResult(adata=adata, source_meta=source_meta)

    except json.JSONDecodeError:
        raise RuntimeError(
            f"R script output was not valid JSON.\nstdout: {proc.stdout}\nstderr: {proc.stderr}"
        )
    finally:
        # Only clean up if we don't need partial_state
        if os.path.isdir(tmp_dir) and not any(
            os.path.isfile(os.path.join(tmp_dir, f)) for f in ["matrix.mtx", "matrix.mtx.gz"]
        ):
            shutil.rmtree(tmp_dir, ignore_errors=True)


def read_sce(path: str, decisions: Optional[dict] = None) -> ReaderResult:
    """Read a SingleCellExperiment RDS file via R subprocess."""
    decisions = decisions or {}

    rscript = shutil.which("Rscript")
    if rscript is None:
        return _r_fallback_instructions(path, "SingleCellExperiment")

    r_script = str(_SCRIPTS_DIR / "sce_reader.R")
    tmp_dir = tempfile.mkdtemp(prefix="stanobj_sce_")

    cmd = [rscript, r_script, path, tmp_dir]
    assay_decision = decisions.get("assay_selection")
    if assay_decision:
        cmd.extend(["--assay", assay_decision])

    try:
        proc = subprocess.run(
            cmd, capture_output=True, text=True, timeout=600
        )

        stdout = proc.stdout.strip()
        if not stdout:
            raise RuntimeError(f"R script produced no output. stderr: {proc.stderr}")

        result_json = json.loads(stdout)

        if result_json.get("status") == "error":
            raise RuntimeError(result_json["message"])

        if result_json.get("status") == "decision_needed":

                options=result_json["options"],
                recommendation=result_json.get("recommendation"),
                reason=result_json.get("reason"),

            if fname.startswith("reduction_") and fname.endswith(".csv"):
                red_name = fname.replace("reduction_", "").replace(".csv", "")
                emb = pd.read_csv(os.path.join(tmp_dir, fname), index_col=0)
                key = f"X_{red_name}"
                if emb.shape[0] == adata.n_obs:
                    adata.obsm[key] = emb.values

        slot_hint = result_json.get("slot_hint", "unknown")
        hint_map = {"counts": "counts", "logcounts": "log1p", "normcounts": "normalized"}
        matrix_type_hint = hint_map.get(slot_hint, None)

        source_meta = {
            "source_format": "sce",
            "reader_used": "sce_reader",
            "matrix_orientation_before": "genes_x_cells",
            "transposed": True,
            "assay_selected": result_json.get("assay"),
            "raw_counts_found": slot_hint == "counts",
            "matrix_type_hint": matrix_type_hint,
            "decompressed": False,
            "warnings": [],
        }

        return ReaderResult(adata=adata, source_meta=source_meta)

    except json.JSONDecodeError:
        raise RuntimeError(
            f"R script output was not valid JSON.\nstdout: {proc.stdout}\nstderr: {proc.stderr}"
        )
    finally:
        if os.path.isdir(tmp_dir):
            has_mtx = any(
                os.path.isfile(os.path.join(tmp_dir, f))
                for f in ["matrix.mtx", "matrix.mtx.gz"]
            )
            if not has_mtx:
                shutil.rmtree(tmp_dir, ignore_errors=True)

#!/usr/bin/env python3
"""
stanobj — Standardize heterogeneous single-cell data into canonical h5ad.

Exit codes:
  0  = success, outputs written
  1  = fatal error
  10 = decision needed (structured JSON on stdout)
"""

import argparse
import json
import os
import sys
import traceback
from pathlib import Path

# Ensure scripts/ is on the path for imports
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from readers.base import DecisionNeeded, ReaderResult
from detection import detect_format, compute_evidence, classify_with_override
from validation import validate_adata
from standardize import (
    standardize_obs,
    standardize_var,
    standardize_obsm,
    assign_layers,
    add_provenance,
    ensure_sparse,
)
from report import generate_report_json, generate_audit_log, write_outputs
from utils import (
    is_archive,
    extract_archive,
    decompress_to_temp,
    strip_compression_ext,
    sample_matrix,
    cleanup_temp,
)


def parse_args():
    parser = argparse.ArgumentParser(
        description="Standardize single-cell data into canonical h5ad",
        prog="stanobj",
    )
    parser.add_argument("input", help="Input file or directory")
    parser.add_argument("-o", "--output", required=True, help="Output .h5ad path")
    parser.add_argument("--format", dest="force_format", default=None,
                        help="Override format detection")
    parser.add_argument("--decision", action="append", default=[],
                        help="Pre-supply decisions as key=value (repeatable)")
    return parser.parse_args()


def parse_decisions(decision_args: list[str]) -> dict:
    """Parse --decision key=value args into a dict."""
    decisions = {}
    for arg in decision_args:
        if "=" not in arg:
            print(f"Warning: ignoring malformed --decision '{arg}' (expected key=value)",
                  file=sys.stderr)
            continue
        key, value = arg.split("=", 1)
        decisions[key] = value
    return decisions


def resolve_input(path: str) -> tuple[str, bool, list[str]]:
    """
    Handle archives and compression.

    Returns:
        (resolved_path, was_decompressed, temp_paths_to_cleanup)
    """
    temps = []

    if is_archive(path):
        dest = extract_archive(path)
        temps.append(dest)
        # Look for recognizable content inside
        contents = list(Path(dest).rglob("*"))
        # If there's a single subdirectory, descend into it
        top_items = [p for p in Path(dest).iterdir()]
        if len(top_items) == 1 and top_items[0].is_dir():
            return str(top_items[0]), True, temps
        return dest, True, temps

    # Single compressed file that needs seekable access
    p = Path(path)
    stripped, was_compressed = strip_compression_ext(path)
    stripped_ext = Path(stripped).suffix.lower()

    if was_compressed and stripped_ext in (".h5", ".hdf5", ".h5ad", ".rds"):
        tmp_path = decompress_to_temp(path)
        temps.append(tmp_path)
        return tmp_path, True, temps

    # For .csv.gz, .tsv.gz, .mtx.gz — readers handle natively
    return path, was_compressed, temps


def load_source(path: str, fmt: str, decisions: dict) -> ReaderResult:
    """Dispatch to the appropriate reader module."""
    if fmt == "mtx":
        from readers.mtx_reader import read_mtx
        return read_mtx(path, decisions)
    elif fmt in ("csv", "tsv"):
        from readers.csv_reader import read_csv
        delimiter = "\t" if fmt == "tsv" else ","
        return read_csv(path, delimiter=delimiter, decisions=decisions)
    elif fmt == "10x_h5":
        from readers.h5_reader import read_10x_h5
        return read_10x_h5(path, decisions)
    elif fmt == "generic_h5":
        from readers.h5_reader import read_generic_h5
        return read_generic_h5(path, decisions)
    elif fmt == "h5ad":
        from readers.h5ad_reader import read_h5ad
        return read_h5ad(path, decisions)
    elif fmt == "loom":
        from readers.loom_reader import read_loom
        return read_loom(path, decisions)
    elif fmt == "seurat_rds":
        from readers.r_bridge import read_seurat
        return read_seurat(path, decisions)
    elif fmt == "sce":
        from readers.r_bridge import read_sce
        return read_sce(path, decisions)
    else:
        raise ValueError(f"Unsupported format: {fmt}")


def run(args):
    decisions = parse_decisions(args.decision)
    all_warnings = []
    temps = []

    try:
        # 1. Resolve input (decompress / extract)
        resolved_path, decompressed, temps = resolve_input(args.input)

        # 2. Detect format
        if args.force_format:
            fmt = args.force_format
        else:
            fmt = detect_format(resolved_path)

        # 3. Read source
        reader_result = load_source(resolved_path, fmt, decisions)
        adata = reader_result.adata
        source_meta = reader_result.source_meta
        source_meta["source_path"] = os.path.abspath(args.input)
        source_meta["decompressed"] = decompressed
        if decompressed:
            source_meta["original_filename"] = Path(args.input).name
        all_warnings.extend(source_meta.get("warnings", []))

        n_cells_before = adata.n_obs
        n_genes_before = adata.n_vars

        # 4. Classify matrix type
        X_sample = sample_matrix(adata.X, n_sample=1000, seed=42)
        evidence = compute_evidence(X_sample)
        matrix_classification = classify_with_override(
            evidence,
            source_hint=source_meta.get("matrix_type_hint"),
        )
        all_warnings.extend(matrix_classification.get("warnings", []))

        # Check if confidence is too low and no decision supplied
        if (matrix_classification["confidence"] == "low"
                and matrix_classification["matrix_type"] == "unknown"
                and "matrix_type" not in decisions):
            raise DecisionNeeded(
                decision_type="matrix_type",
                context=(
                    f"Cannot confidently classify matrix type.\n"
                    f"Evidence: {json.dumps(evidence, indent=2)}"
                ),
                options=["counts", "normalized", "log1p", "scaled", "unknown"],
                recommendation="unknown",
                reason="Heuristic confidence is low",
            )

        if "matrix_type" in decisions:
            matrix_classification["matrix_type"] = decisions["matrix_type"]
            matrix_classification["matrix_type_source"] = "user_decision"

        matrix_type = matrix_classification["matrix_type"]

        # 4b. Check for mixed modalities
        if "feature_type" in adata.var.columns:
            ft_unique = list(adata.var["feature_type"].unique())
            if len(ft_unique) > 1:
                modality_decision = decisions.get("modality_filter")
                if modality_decision is None:
                    rna_count = int((adata.var["feature_type"] == "Gene Expression").sum())
                    non_rna_count = adata.n_vars - rna_count
                    raise DecisionNeeded(
                        decision_type="modality_filter",
                        context=(
                            f"Mixed feature types detected: {ft_unique}. "
                            f"{rna_count} Gene Expression features, "
                            f"{non_rna_count} non-RNA features."
                        ),
                        options=["rna_only", "keep_all"],
                        recommendation="rna_only",
                        reason="Default workflow is scRNA-seq standardization",
                    )
                if modality_decision == "rna_only":
                    rna_mask = adata.var["feature_type"] == "Gene Expression"
                    n_before = adata.n_vars
                    adata = adata[:, rna_mask].copy()
                    source_meta["rna_only_subset_applied"] = True
                    source_meta["modality_filter"] = "rna_only"
                    all_warnings.append(
                        f"Filtered to RNA-only: {adata.n_vars} of {n_before} features kept"
                    )
                else:
                    source_meta["modality_filter"] = "keep_all"
                    all_warnings.append("Mixed modalities kept — feature_type column preserved")

        # 5. Standardize
        dataset_name = Path(args.input).stem

        # obs
        obs_unique_needed = len(set(adata.obs_names)) != len(adata.obs_names)
        adata.obs = standardize_obs(
            adata.obs, dataset_name=dataset_name, make_unique=obs_unique_needed
        )
        if obs_unique_needed:
            source_meta["obs_name_strategy"] = "uniquified"
            all_warnings.append("obs_names were not unique — made unique")

        # var
        var_result = standardize_var(adata.var, return_rename_map=True)
        adata.var = var_result[0]
        rename_map = var_result[1]
        if rename_map:
            source_meta["var_name_strategy"] = "dedup_with_suffix"
            source_meta["duplicate_genes_resolved"] = len(rename_map)
            all_warnings.append(
                f"{len(rename_map)} duplicate gene names resolved with suffix"
            )

        # obsm
        if adata.obsm:
            adata.obsm = standardize_obsm(dict(adata.obsm))

        # layers
        source_layers = dict(adata.layers) if adata.layers else {}
        adata = assign_layers(adata, matrix_type=matrix_type, source_layers=source_layers)

        # sparsity
        converted = ensure_sparse(adata, threshold=10_000)
        if converted:
            all_warnings.append("Dense matrix converted to sparse (n_obs >= 10000)")

        # provenance
        adata = add_provenance(
            adata, source_meta, matrix_classification, decisions, all_warnings
        )

        # 6. Validate
        validation = validate_adata(adata, matrix_type=matrix_type)
        all_warnings.extend(validation.warnings)

        if not validation.passed:
            print(json.dumps({
                "status": "error",
                "fatal_errors": validation.fatal_errors,
                "warnings": all_warnings,
            }, indent=2))
            return 1

        # Update provenance warnings with validation warnings
        adata.uns["stanobj"]["warnings"] = all_warnings

        # 7. Generate reports and write
        x_contents = adata.uns["stanobj"].get("x_contents", "unknown")
        source_meta["x_contents"] = x_contents
        source_meta["layers_written"] = list(adata.layers.keys())
        source_meta["embeddings_preserved"] = list(adata.obsm.keys())
        source_meta["raw_counts_found"] = "counts" in adata.layers
        source_meta["feature_types_present"] = (
            list(adata.var["feature_type"].unique())
            if "feature_type" in adata.var.columns else []
        )

        report_json = generate_report_json(
            source_path=os.path.abspath(args.input),
            source_format=source_meta.get("source_format", "unknown"),
            source_meta=source_meta,
            matrix_classification=matrix_classification,
            n_cells_before=n_cells_before,
            n_genes_before=n_genes_before,
            n_cells_after=adata.n_obs,
            n_genes_after=adata.n_vars,
            decisions=decisions,
            warnings=all_warnings,
        )

        audit_log = generate_audit_log(
            source_path=os.path.abspath(args.input),
            source_format=source_meta.get("source_format", "unknown"),
            output_path=os.path.abspath(args.output),
            source_meta=source_meta,
            matrix_classification=matrix_classification,
            x_contents=x_contents,
            n_cells=adata.n_obs,
            n_genes=adata.n_vars,
            decisions=decisions,
            warnings=all_warnings,
        )

        # Ensure output directory exists
        os.makedirs(Path(args.output).parent, exist_ok=True)

        write_outputs(adata, args.output, report_json, audit_log)

        return 0

    except DecisionNeeded as e:
        print(e.to_json())
        return 10

    except Exception as e:
        print(json.dumps({
            "status": "error",
            "message": str(e),
            "traceback": traceback.format_exc(),
        }, indent=2), file=sys.stderr)
        return 1

    finally:
        for tmp in temps:
            cleanup_temp(tmp)


def main():

# stanobj — Standardize Single-Cell Data to Canonical h5ad

Convert heterogeneous single-cell transcriptomics datasets into canonical,
analysis-ready `.h5ad` files with full provenance tracking.

## Supported Formats

| Format | Extension | Reader |
|--------|-----------|--------|
| Matrix Market / 10x triplet | `.mtx`, directory | mtx_reader |
| CSV expression matrix | `.csv`, `.csv.gz` | csv_reader |
| TSV expression matrix | `.tsv`, `.tsv.gz` | csv_reader |
| 10x Genomics HDF5 | `.h5` | h5_reader |
| Generic HDF5 | `.h5`, `.hdf5` | h5_reader |
| Existing h5ad | `.h5ad` | h5ad_reader |
| Loom | `.loom` | loom_reader |
| Seurat RDS | `.rds` | seurat_reader (R subprocess) |
| SingleCellExperiment RDS | `.rds` | sce_reader (R subprocess) |

Archives supported: `.tar.gz`, `.tgz`, `.tar.bz2`, `.tar`, `.zip`
Compression: `.gz`, `.bz2` (transparent for most formats)

## Workflow

1. Identify the input file/directory from the user's request
2. Ask the user for an output path (required: `-o <path.h5ad>`)
3. Run the orchestrator:

```bash
python ~/.claude/skills/stanobj/scripts/stanobj.py <input> -o <output.h5ad>
```

4. Check the exit code:
   - **Exit 0**: Success. Read the audit log and present a summary to the user.
   - **Exit 10**: Decision needed. Parse the JSON from stdout, present options to the user (or decide if obvious), then re-invoke with `--decision <type>=<choice>`.
   - **Exit 1**: Fatal error. Read stderr and report to user.

5. After success, optionally suggest running `scrna-reader` to verify the output.

## Decision Protocol

When the orchestrator cannot proceed autonomously, it exits with code 10 and
prints a JSON decision request to stdout:

```json
{
  "status": "decision_needed",
  "decision_type": "assay_selection",
  "context": "Seurat object has 3 assays: RNA, SCT, ADT",
  "options": ["RNA", "SCT", "ADT"],
  "recommendation": "RNA",
  "reason": "RNA has the most features and contains raw counts"
}
```

Re-invoke with: `--decision assay_selection=RNA`

Multiple decisions can be supplied: `--decision assay_selection=RNA --decision matrix_type=counts`

### Known Decision Points

| Type | Trigger | Options |
|------|---------|---------|
| `format_detection` | Unrecognized file format | format names |
| `assay_selection` | Multiple assays in Seurat/SCE | assay names |
| `matrix_orientation` | Ambiguous CSV/TSV layout | `cells_x_genes`, `genes_x_cells` |
| `matrix_type` | Low-confidence classification | `counts`, `normalized`, `log1p`, `scaled`, `unknown` |
| `layer_selection` | Multiple candidate layers | layer names |
| `modality_filter` | Mixed RNA + non-RNA features | `rna_only`, `keep_all` |

```markdown
# Format-Specific Notes and Gotchas

## CSV / TSV
- Most ambiguous format. Orientation detection uses label pattern matching and shape heuristics.
- First column is assumed to be row identifiers (index_col=0).
- Gene-like patterns: ENSG..., uppercase symbols (TP53), title-case (Trp53).
- Barcode-like patterns: ACGT sequences, prefix-number patterns.
- If neither rows nor columns match known patterns, the orchestrator requests a decision.

## Matrix Market (MTX)
- Convention: genes x cells. Always transposed to cells x genes.
- Look for sibling files: barcodes.tsv[.gz], features.tsv[.gz] or genes.tsv[.gz].
- features.tsv may have 2 columns (id, name) or 3 columns (id, name, feature_type).
- 3rd column indicates modality: Gene Expression, Antibody Capture, CRISPR Guide Capture.

## 10x HDF5 (.h5)
- Contains `matrix/` group with CSC sparse components.
- Internal format is genes x cells. Transposed by reader.
- Features group may include feature_type for multi-modal data.

## Generic HDF5
- No assumed structure. Reader inspects the HDF5 tree.
- If `matrix/` group found → treated as 10x.
- If `X/` or `obs/` found → treated as h5ad-like.
- Otherwise → decision request with tree dump.

## Existing h5ad
- NOT assumed to be canonical.
- Full pipeline runs: validation, standardization, type classification.
- Useful for re-standardizing h5ad files from different pipelines.

## Loom
- Convention: genes x cells (rows = genes, columns = cells).
- Always transposed by reader.
- row_attrs → var columns, col_attrs → obs columns.
- "Gene" or "gene" row attribute used for var_names.
- "CellID" or "cellid" col attribute used for obs_names.

## Seurat RDS
- May contain multiple assays: RNA, SCT, ADT, integrated, etc.
- Do NOT assume default assay is RNA.
- Critical slot distinctions: counts = raw, data = log-normalized, scale.data = scaled.
- Exported via R subprocess to temp MTX directory + metadata CSV.
- If RDS contains a SingleCellExperiment instead, automatically redirected.

## SingleCellExperiment RDS
- May contain multiple assays: counts, logcounts, normcounts.
- Exported via R subprocess to temp MTX directory + metadata CSV.
- altExp objects may contain non-RNA modalities (not exported by default).

## R Fallback
- If Rscript is not available, the skill provides manual R commands for conversion.
- Users can run the provided R code, then re-invoke stanobj on the resulting files.

# Decision Points Reference

When the orchestrator encounters ambiguity it cannot resolve confidently, it exits
with code 10 and emits a JSON decision request.

## format_detection
**Trigger:** File format cannot be determined from extension or content.
**Options:** Format names (csv, tsv, mtx, 10x_h5, h5ad, loom, seurat_rds, sce).
**Context:** File path and content summary.

## assay_selection
**Trigger:** Seurat or SCE object has multiple assays.
**Options:** Available assay names.
**Context:** Assay inventory with feature counts and available slots.
**Recommendation:** Default assay (Seurat) or "counts" assay (SCE) if present.

## matrix_orientation
**Trigger:** CSV/TSV with ambiguous row/column labels.
**Options:** `cells_x_genes`, `genes_x_cells`
**Context:** Matrix shape, sample row/column labels.

## matrix_type
**Trigger:** Cannot confidently classify the main matrix.
**Options:** `counts`, `normalized`, `log1p`, `scaled`, `unknown`
**Context:** Evidence metrics (has_negatives, is_integer, max_value, sparsity, row_sums_cv).

## layer_selection
**Trigger:** Multiple candidate layers in h5ad, unclear which is primary.
**Options:** Layer names.
**Context:** Available layers with detected types.

## modality_filter
**Trigger:** Feature types include non-RNA modalities (ADT, CRISPR, Peaks).
**Options:** `rna_only`, `keep_all`
**Context:** Feature type distribution.
